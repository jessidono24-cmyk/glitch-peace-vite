## Technical Implementation

### Mode Manager System
```javascript
// src/modes/mode-manager.js
class ModeManager {
  constructor() {
    this.currentMode = 'grid'; // default
    this.modes = {
      grid: new GridMode(),
      shooter: new ShooterMode(),
      rpg: new RPGMode(),
      constellation: new ConstellationMode(),
      platformer: new PlatformerMode(),
      rhythm: new RhythmMode()
    };
  }
  
  switchMode(modeName) {
    // Clean up current mode
    this.modes[this.currentMode].cleanup();
    
    // Initialize new mode
    this.currentMode = modeName;
    this.modes[modeName].init();
  }
  
  update(dt) {
    this.modes[this.currentMode].update(dt);
  }
  
  render(ctx) {
    this.modes[this.currentMode].render(ctx);
  }
}
```
